#mod_rewrite starts here (url remove.html) RewriteEngine on # Does not apply to existing directories, meaning that if the folder exists on the server then don't change anything and don't run the rule! RewriteCond %{REQUEST_FILENAME} !-d # Check for file in directory with .html extension RewriteCond %{REQUEST_FILENAME}\.html -f # Here we acutally show the page that have the .html extension, grabbing entire RewriteRule ^(.*)$ $1.html [NC,L]
